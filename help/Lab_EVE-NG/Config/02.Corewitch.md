1.	Đặt tên cho thiết bị và tạo mới vlan
conf t
hostname Core_Switch
!
Vlan 99
Vlan 101-102
exit

2.	Đặt địa chỉ ip cho port E0/0
Conf t
interface Ethernet0/0
 no switchport
 ip address 10.10.10.2 255.255.255.0
 no shut
exit

3.	Đặt địa chỉ ip cho các interface vlan 99,  101, 102
interface Vlan99
 ip address 10.99.0.254 255.255.255.0
no shut
!
interface Vlan101
 ip address 10.101.0.254 255.255.255.0
no shut
!
interface Vlan102
 ip address 10.102.0.254 255.255.255.0
no shut
!

4.	Cấu hình mode trunk cho port E1/1 và E1/2
interface Ethernet1/1
 switchport trunk encapsulation dot1q
 switchport mode trunk
 exit
!
interface Ethernet1/2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 exit

5.	Cấu hình mode Access cho port E1/3
interface Ethernet1/3
 switchport access vlan 99
 switchport mode access
exit

6.	Cấu hình Default Route trỏ về Firewall

ip route 0.0.0.0 0.0.0.0 10.10.10.1

7.	Cấu hình DHCP Relay Agent cho các interface (trỏ về DHCP trong Server Farm)
interface Vlan102
ip helper-address 10.99.0.11
exit
